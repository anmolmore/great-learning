>> np.linspace(2.0, 3.0, num=5)<br/>
array([2.  , 2.25, 2.5 , 2.75, 3.  ])

>> np.linspace(2.0, 3.0, num=5, endpoint=False)<br/>
array([2. ,  2.2,  2.4,  2.6,  2.8])

>> np.linspace(2.0, 3.0, num=5, retstep=True)<br/>
(array([2.  ,  2.25,  2.5 ,  2.75,  3.  ]), 0.25)
